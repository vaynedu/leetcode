121. 买卖股票最佳时机

贪心：在最低价买入，在最高价卖出

function maxProfit(prices) {
  let minP = prices[0], maxP = 0;
  for (let i = 0; i < prices.length; i++) {
    minP = Math.min(minP, prices[i]);
    maxP = Math.max(maxP, prices[i] - minP);
  }
  return maxP;
}
