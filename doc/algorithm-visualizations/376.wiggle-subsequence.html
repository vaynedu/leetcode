376. 摆动序列

贪心：统计连续上升/下降段的数量，结果=段数+1

摆动序列
function wiggleMaxLength(nums) {
  if (nums.length < 2) return nums.length;
  let up=1, down=1;
  for (let i=1; i<nums.length; i++) {
    if (nums[i] > nums[i-1]) up = down + 1;
    else if (nums[i] < nums[i-1]) down = up + 1;
  }
  return Math.max(up, down);
}
