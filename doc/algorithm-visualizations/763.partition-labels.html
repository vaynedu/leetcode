763. 划分字母区间

贪心：记录每字符最后出现位置，扩展段直到首尾相遇

当前字符: -
段起点: -
段终点: -
段长度: -
点击「下一步」开始
function partitionLabels(s) {
  const last = {};
  for (let i=0; i<s.length; i++) last[s[i]] = i;
  let start=0, end=0, res=[];
  for (let i=0; i<s.length; i++) {
    end = Math.max(end, last[s[i]]);
    if (i === end) {
      res.push(end - start + 1);
      start = end + 1;
    }
  }
  return res;
}
