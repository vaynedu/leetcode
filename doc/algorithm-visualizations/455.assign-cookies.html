455. 分发饼干

贪心：用最小饼干满足每个孩子，优先满足胃口小的

当前饼干: -
当前孩子: -
已满足: 0 / ${3}
胃口 g[]
饼干 s[]
function findContentChildren(g, s) {
  g.sort((a,b) => a - b);
  s.sort((a,b) => a - b);
  let i = 0, j = 0, count = 0;
  while (i < g.length && j < s.length) {
    if (s[j] >= g[i]) {
      count++; i++; j++;
    } else {
      j++;
    }
  }
  return count;
}
