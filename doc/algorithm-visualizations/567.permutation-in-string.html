Go 代码模板

找到后立即返回 true，效率更高

func checkInclusion(s1, s2 string) bool {
    need := [26]int{}
    for _, c := range s1 {
        need[c-'a']++
    }

    diff := 0  // 还有几个字母未匹配

    for i := range s2 {
        // 入窗口
        idx := s2[i] - 'a'
        if need[idx] > 0 {
            diff++
        }
        need[idx]--

        // 出窗口
        if i >= len(s1) {
            outIdx := s2[i-len(s1)] - 'a'
            if need[outIdx] >= 0 {
                diff--
            }
            need[outIdx]++
        }

        if diff == 0 {
            return true  // 找到即返回，不用继续
        }
    }
    return false
}
