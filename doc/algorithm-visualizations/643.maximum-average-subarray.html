Go 代码模板

固定窗口三步：先算初始窗口 → 滑动（减左边加右边）→ O(1) 更新答案

func findMaxAverage(nums []int, k int) float64 {
    // 1. 先算第一个窗口的和
    sum := 0
    for i := 0; i < k; i++ {
        sum += nums[i]
    }
    maxAvg := float64(sum) / float64(k)

    // 2. 滑动窗口：减左边，加右边
    for i := k; i < len(nums); i++ {
        sum += nums[i]       // 加右边（新元素）
        sum -= nums[i-k]     // 减左边（旧元素）
        avg := float64(sum) / float64(k)
        maxAvg = math.Max(maxAvg, avg)
    }
    return maxAvg
}
