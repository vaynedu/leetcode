53. 最大子数组和

Kadane 算法 — 一次遍历找到最大和

数组
dp[i] — 以 i 结尾的最大子数组和
function maxSubArray(nums) {
  let cur = nums[0], max = nums[0];
  for (let i = 1; i < nums.length; i++) {
    // 要么扩展前一段，要么从 i 重新开始
    cur = Math.max(cur + nums[i], nums[i]);
    max = Math.max(max, cur);
  }
  return max;
}
