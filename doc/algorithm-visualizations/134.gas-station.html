贪心：如果 total < 0 无解；否则从 0 开始，累加负则换起点

function canCompleteCircuit(gas, cost) {
  let total=0, tank=0, start=0;
  for (let i=0; i<gas.length; i++) {
    total += gas[i] - cost[i];
    tank += gas[i] - cost[i];
    if (tank < 0) { start = i+1; tank = 0; }
  }
  return total < 0 ? -1 : start;
}
